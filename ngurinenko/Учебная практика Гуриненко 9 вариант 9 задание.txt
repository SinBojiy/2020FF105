packagecom.company;// Стандартный пакет
importjava.util.Scanner; // Подключаем сканер
publicclassMain {

publicstaticvoidmain(String[] args) {
System.out.println("Введите строку: "); // Запрашиваем строку у пользователя
Scannerscan = newScanner(System.in); // С помощью сканера считываем с клавиатуры и заносим в переменную str
Stringstr = scan.nextLine();
intsum=0,a=0;// Переменные для подсчета
a=str.indexOf("abc",0);// делаем поиск в строке начиная с нулевого индекса, и записываем индекс вхождения в a.
if(a!=-1) {// если Abcнашлость в тексте, то
sum = ++sum;// к sum добавляем 1
}
while(a!=-1){ // пока indexof не вернет нам -1(тоесть "abc" не найдено), выполняем поиск
a=str.indexOf("abc",a+1);// также делаем поиск по строке, но ищем начиная с a+1(Чтобы поиск не зациклился на одноном выражении)
if(a!=-1) {
sum = ++sum;
}
        }
System.out.print("Группа букв abc входит в строку: " + sum + " раз");// Выводим информацию
}
}
